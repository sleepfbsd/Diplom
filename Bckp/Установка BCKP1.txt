#Общая директория для файлов

sudo chmod 755 /srv/mediawiki-images

# sudo nano /etc/exports

#Создание резервного копирования базы sql

sudo useradd -u 113 -g 113 -m -s /bin/bash postgres
sudo mkdir -p /srv/backups/postgresql
sudo chown postgres:postgres /srv/backups/postgresql   
sudo chmod 755 /srv/backups/postgresql

sudo nano /etc/exports

/srv/backups/postgresql 10.129.0.0/24(rw,sync,no_subtree_check)
